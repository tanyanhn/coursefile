\begin{pro}
  What is the inverse of the following matrix?
  \begin{displaymath}
    A =
    \begin{bmatrix}
      1 & 0 & 0\\
      1 & -1 & 0 \\
      1 & -2 & 1
    \end{bmatrix}
  \end{displaymath}
\end{pro}

\begin{sol}
  \begin{displaymath}
    A^{-1} =
    \begin{bmatrix}
            1 & 0 & 0\\
      1 & -1 & 0 \\
      1 & -2 & 1
    \end{bmatrix}
  \end{displaymath}
\end{sol}
