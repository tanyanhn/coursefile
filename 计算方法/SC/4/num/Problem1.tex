\begin{pro}
  \begin{itemize}
  \item[(a)]
    What is the degree of Simpson's rule for numerical quadrature?

  \item[(b)]
    What is the degree of an $n$-point Gaussian quadrature rule?
  \end{itemize}
\end{pro}

\begin{sol}
  \begin{itemize}
  \item[(a)]
    $3$.

  \item[(b)]
    $2n-1$.
  \end{itemize}
\end{sol}
