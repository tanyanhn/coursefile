\begin{pro}
  Rank the following types of quadrature rules
  in order of their degree for the same number of nodes
  ($1$ for highest degree, etc.):
  \begin{itemize}
  \item[(a)]
    Newton-Cotes

  \item[(b)]
    Gaussian

  \item[(c)]
    Kronrod
  \end{itemize}
\end{pro}

\begin{sol}
  1. Gaussian; 2. Kronrod; 3. Newton-Cotes.
\end{sol}
