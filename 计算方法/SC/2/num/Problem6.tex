\begin{pro}
  What are the eigenvalues and corresponding eigenvectors of the
  following matrix?
  \begin{displaymath}
    \begin{bmatrix}
      1 & 2 & -4 \\
      0 & 2 & 1 \\
      0 & 0 & 3
    \end{bmatrix}
  \end{displaymath}
\end{pro}

\begin{sol}
  \begin{align*}
    \lambda_1 &= 1, \quad \mathbf{x}_1 =c
    \begin{bmatrix}
      1 & 0 & 0
    \end{bmatrix}^T; \\
    \lambda_2 &= 2, \quad \mathbf{x}_2 =c
                \begin{bmatrix}
                  2 & 1 & 0
                \end{bmatrix}^T; \\
    \lambda_3 &= 3, \quad \mathbf{x}_3 = c
              \begin{bmatrix}
                -1 & 1 & 1
              \end{bmatrix}^T,
  \end{align*}
  where $c$ is an arbitrary constant.
\end{sol}
